        PLOT_XY_LC.setOnScroll(new EventHandler<ScrollEvent>() {
            @Override
            public void handle(ScrollEvent ev) {
                if (ZoomX) {
                    double zoomFactor = 1.05;
                    double deltaY = ev.getDeltaY();

                    if (deltaY < 0) {
                        zoomFactor = 2 - zoomFactor;
                    }

                    NumberAxis xAxisLocal = ((NumberAxis) PLOT_XY_LC.getXAxis());
                    xAxisLocal.setUpperBound(xAxisLocal.getUpperBound() * zoomFactor);
                    xAxisLocal.setLowerBound(xAxisLocal.getLowerBound() * zoomFactor);
                    xAxisLocal.setTickUnit(1);
                    xAxisLocal.setMinorTickCount(10);
                }
                if (ZoomY) {
                    double zoomFactor = 1.05;
